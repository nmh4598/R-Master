1. Programmer une fonction cr qui centre et r√©duit un tableau de donn√©es.

ecartype = function(x){
  sqrt(mean(x^2)-(mean(x))^2)
}
cr <- function(X){
  n <- nrow(X)
  moyennes <- apply(X, FUN = mean, MARGIN = 2)
  moyennes <- matrix(rep(moyennes, n), nrow = n, byrow = T)
  ectypes <- apply(X, FUN = ecartype, MARGIN = 2)
  ectypes <- matrix(rep(ectypes,n), nrow = n, byrow = T)
  return((X-moyennes)/ectypes)
}

2. Programmer une fonction acp qui r√©alise les calculs de l‚ÄôACP (on donnera en entr√©e le tableau centr√© r√©duit)

acp_main<- function(Z){
  n <- nrow(Z)
  Z <- cr(Z)  #Z est le tableau centr√©-reduit du tableau entr√©
  Z <- as.matrix(Z,nrow=n,byrow=T)
  b<- t(Z)%*%Z/nrow(Z)   #Matrice de corr√©lation de Z
  c<-eigen(b)
  d<- c$values  #les valeur propres
  e <- c$vectors #les vecteurs propres
  inertie<-d/sum(d) # l‚Äôinertie axe par axe (en pourcentage)
  Coord_Ind <- Z%*%e # coordonn√©es des individus sur les axes (pour 3 axes)
  Coord_Var=e*(rep(1,ncol(Z))%*%sqrt(t(eigen(b)$values))) # coordonn√©es des variables sur les axes (pour 3 axes)
  rownames(Coord_Var)=colnames(Z)
  for (i in seq(0,ncol(Z),2)){ # il y a sens oppos√© avec PCA en FactomineR au colonne 2 et 4 donc, nous changeons les signes, nous expliquerons en d√©tail dans la comparaison avec les r√©sultats de l'ACP 
    Coord_Ind[,i]=-Coord_Ind[,i]
    Coord_Var[,i]=-Coord_Var[,i]
  }
  a <- list(eig=inertie,ind=Coord_Ind,var=Coord_Var)
  return(a)
}

3. Programmer une fonction plotacp qui r√©alise les graphiques de l‚ÄôACP (on donnera en entr√©e la liste contenant les calculs de l‚ÄôACP et les axes choisis et le type de graphique, ‚Äúvariables‚Äù ou ‚Äúindividus‚Äù)

plotacp <-function(don,choix,axe1,axe2){
  ACP_main_don<-acp_main(don)
  nom_axe1 <- paste("Dim",axe1,"(",round(ACP_main_don$eig[axe1]*100,2),"%)",sep=" ")
  nom_axe2 <- paste("Dim",axe2,"(",round(ACP_main_don$eig[axe2]*100,2),"%)",sep=" ")
  if (choix=='ind'){
    plot(ACP_main_don$ind[,axe1],ACP_main_don$ind[,axe2],lwd=2,ylim=c(-4.5,3),main="PCA graph of individuals",xlab=nom_axe1,ylab=nom_axe2)
    abline(h = 0,lty=2)
    abline(v = 0,lty=2)
    text(ACP_main_don$ind[,axe1],ACP_main_don$ind[,axe2],rownames(ACP_main_don$ind),pos=1)
  } else{
    plot(ACP_main_don$var[,axe1],ACP_main_don$var[,axe2],lwd=2,main="PCA graph of variables",xlab=nom_axe1,ylab=nom_axe2,xlim=c(-1,1),ylim=c(-1,1),pch=16,asp=1)
    
    radius <- 1
    theta <- seq(0, 2 * pi, length = 200)
    lines(x = radius * cos(theta), y = radius * sin(theta))

    s <- seq(length(ACP_main_don$var[,axe1])) 
    arrows(0,0, ACP_main_don$var[s,axe1], ACP_main_don$var[s,axe2])
    text(ACP_main_don$var[,axe1],ACP_main_don$var[,axe2],rownames(ACP_main_don$var),pos=4)
    abline(h=0,lty=2)
    abline(v=0,lty=2)
  }
}

4. Programmer une fonction indsup qui calcule les coordonn√©es des individus suppl√©mentaires (pour 3 axes).

# La projection des individus suppl√©mentaires  est une fonctionnalit√© importante de l‚Äôanalyse en composantes principales. L‚Äôobjectif est de positionner de nouveaux individus par rapport √† ceux  qui ont contribu√© √† la construction du rep√®re factoriel.

indsup <- function(newind,X){     #newind est le(s) individu(s) suppl√©metaire(s), X est le tableau initial
  nbnewind = length(newind[,1])   
  mean=t(matrix(apply(X,2,mean),nrow=ncol(X),ncol=nbnewind)) #Matrice concernant les moyennes des colonnes de l'ancien tableau X, avec la dimension identique √† la matrice newind. Elle nous aidera √† cr√©er un tableau centr√© r√©duit de newind.
  ectype=t(matrix(apply(X,2,ecartype)*sqrt((nrow(X)-1)/nrow(X)),nrow=ncol(X),ncol=nbnewind)) #identiquement avec ecartype
  newind_cr = (newind-mean)/ectype  #R√©alisons le tableau centr√©-r√©duite pour les individus suppl√©metaires
  Z <-cr(X)    #Nous utilisons le tableau centr√©-r√©duit de X 
  n <- nrow(Z)
  Z <- as.matrix(Z,nrow=n,byrow=T)
  b<- t(Z)%*%Z/nrow(Z) 
  c<-eigen(b)
  e <- c$vectors 
  Coord_Ind <- newind_cr%*%e  #Coordonn√©es des individus suppl√©mentaires, sur nos anciens axes.
  return(Coord_Ind)
}
